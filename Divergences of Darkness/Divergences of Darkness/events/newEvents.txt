	id = 81002211
	title = "81002211.T"
	desc = "81002211.D"
	picture = "treaty"
	
	trigger = {
		war = no
		OR = {
			any_neighbor_country = {
				alliance_with = THIS
				NOT = { vassal_of = THIS }
				NOT = { is_our_vassal = THIS }
				NOT = { in_sphere = THIS }
				NOT = { is_sphere_leader_of = THIS }
			}
			any_greater_power = {
				alliance_with = THIS
				NOT = { is_our_vassal = THIS }
				NOT = { is_sphere_leader_of = THIS }
			}
		}
	}
	
	mean_time_to_happen = {
		months = 360
	}
	
	option = {
		name = "81002211.A"
		prestige = -10
		random_country = {
			limit = {
				alliance_with = THIS
				NOT = { vassal_of = THIS }
				NOT = { is_our_vassal = THIS }
				NOT = { in_sphere = THIS }
				NOT = { is_sphere_leader_of = THIS }
			}
			leave_alliance = THIS
			relation = { who = THIS value = -50 }
		}	
	}
}
